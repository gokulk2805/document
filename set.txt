def solve():
    n = int(input())
    items = list(map(int, input().split()))

    min_diff = float('inf')
    best_group1 = []
    best_group2 = []

    for i in range(1 << n):
        group1 = []
        group2 = []
        for j in range(n):
            if (i >> j) & 1:
                group1.append(items[j])
            else:
                group2.append(items[j])

        sum1 = sum(group1)
        sum2 = sum(group2)
        diff = abs(sum1 - sum2)

        if diff < min_diff:
            min_diff = diff
            best_group1 = group1
            best_group2 = group2

    best_group1.sort(reverse=True)
    best_group2.sort(reverse=True)

    print(best_group1)
    print(best_group2)
    print(min_diff)

solve()

def process_integers(input_string):
  """
  Processes a list of integers, removes duplicates while preserving order,
  concatenates the unique integers into a single string, and prints the result.

  Args:
    input_string: A string of space-separated integers.
  """
  integers = input_string.split()
  unique_integers = []
  for integer in integers:
    if integer not in unique_integers:
      unique_integers.append(integer)
  
  concatenated_string = "".join(unique_integers)
  print(concatenated_string)

# Read the input from the user
input_string = input()

# Process the integers and print the result
process_integers(input_string)

def analyze_words(w1, w2):
    w1 = w1.strip().lower()
    w2 = w2.strip().lower()

    set1 = set(w1)
    set2 = set(w2)

    common_letters = sorted(list(set1.intersection(set2)))
    unique_letters = sorted(list(set1.symmetric_difference(set2)))

    is_superset = set1.issuperset(set2)
    no_common_letters = not bool(common_letters)

    print(common_letters)
    print(unique_letters)
    print(is_superset)
    print(no_common_letters)

w1 = input()
w2 = input()
analyze_words(w1, w2)


FunctionPurposeadd(x)Adds an elementupdate(iterable)Adds multiple elementsremove(x)Removes element (error if absent)discard(x)Removes element (no error)pop()Removes a random elementclear()Removes all elementscopy()Creates a copy of setunion()Combines setsintersection()Common elementsdifference()Elements in first set onlysymmetric_difference()Elements in either set but not bothissubset()Checks subsetissuperset()Checks supersetisdisjoint()Checks no common elementslen(s)Number of elements
max(s)Largest element
min(s)Smallest element
sum(s)Sum of elements
sorted(s)Returns sorted list
