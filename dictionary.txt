	def combine_expense_reports():

	 marketing = {}

	 operations = {}



	 n = int(input())

	 for _ in range(n):

	 category = input()

	 expense = int(input())

	 marketing[category] = expense



	 m = int(input())

	 for _ in range(m):

	 category = input()

	 expense = int(input())

	 operations[category] = expense



	 combined = {}

	 for category, expense in marketing.items():

	 combined[category] = expense



	 for category, expense in operations.items():

	 if category in combined:

	 combined[category] += expense

	 else:

	 combined[category] = expense



	 print("Combined Dictionary:", combined)



	combine_expense_reports()

	def find_smallest_and_largest_order_ids():

	 n = int(input())

	 orders = {}

	 for _ in range(n):

	 order_id = int(input())

	 order_amount = int(input())

	 orders[order_id] = order_amount



	 smallest_key = None

	 smallest_value = float('inf')

	 largest_key = None

	 largest_value = float('-inf')



	 for key, value in orders.items():

	 if value < smallest_value:

	 smallest_value = value

	 smallest_key = key

	 if value > largest_value:

	 largest_value = value

	 largest_key = key



	 print(f"Smallest Order ID: {smallest_key},")

	 print(f"Largest Order ID: {largest_key}")



	find_smallest_and_largest_order_ids()

	def compare_prices():

	 n1 = int(input())

	 dict1 = {}

	 for _ in range(n1):

	 key = int(input())

	 value = int(input())

	 dict1[key] = value



	 n2 = int(input())

	 dict2 = {}

	 for _ in range(n2):

	 key = int(input())

	 value = int(input())

	 dict2[key] = value



	 result = {}



	 for key, value in dict1.items():

	 if key in dict2:

	 result[key] = abs(value - dict2[key])

	 else:

	 result[key] = value



	 for key, value in dict2.items():

	 if key not in dict1:

	 result[key] = value



	 print(result)



	compare_prices()

	def merge_inventories(dict1, dict2):

	 merged_dict = {}

	 for key, value in dict1.items():

	 if key in dict2:

	 merged_dict[key] = value * dict2[key]

	 else:

	 merged_dict[key] = value

	 for key, value in dict2.items():

	 if key not in dict1:

	 merged_dict[key] = value

	 return merged_dict



	n = int(input())

	dict1 = {}

	for _ in range(n):

	 key = int(input())

	 value = int(input())

	 dict1[key] = value



	m = int(input())

	dict2 = {}

	for _ in range(m):

	 key = int(input())

	 value = int(input())

	 dict2[key] = value



	merged_inventory = merge_inventories(dict1, dict2)

	print(merged_inventory)
