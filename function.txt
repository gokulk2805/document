	domestic_rate=5.0

	international_rate=10.0

	remote_rate=15.0

	def calculate_shipping(weight,destination):

	 if weight<0:

	 print("Invalid weight. Weight must be greater than 0.")

	 return None

	 if destination=="Domestic":

	 destination_rate=domestic_rate

	 elif destination=="International":

	 destination_rate=international_rate

	 elif destination=="Remote":

	 destination_rate=remote_rate

	 else:

	 print("Invalid destination.")

	 return None

	 return weight*destination_rate
