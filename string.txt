NUMBERS PALINDROME MESSAGE
s=input()
digit_sum=sum(int(ch) for ch in s if ch.isdigit())
print(digit_sum,end="")
if digit_sum > 9 :
    sum_str=str(digit_sum)
    if sum_str== sum_str[::-1]:
        print(" Palindrome")
    else:
        print(" Not palindrome")


PERFECT VOWEL
s=input().strip()
vowels="aeiou"
if s[0] in vowels:
    print("False")
else:
    p_c= False
    perfect = True
    
    for ch in s :
        if ch in vowels:
            p_c=False
        else:
            if p_c:
                perfect = False
                break
            p_c=True
    if perfect:
        print("True")
    else:
            print("False")

LJUST AND RJUST
n=int(input())
names=[]
prices=[]
for _ in range(n):
    name,price=input().split()
    names.append(name)
    prices.append(price)
    
mname=max(len(name) for name in names)
mprice=max(len(price) for price in prices)

for i  in range(n):
    print(names[i].ljust(mname),prices[i].rjust(mprice))

STR FUNCTIONS 
len(s)Returns length of string
s.upper()Converts to uppercase
s.lower()Converts to lowercase
s.capitalize()First letter uppercase
s.title()First letter of each word uppercase
s.swapcase()Changes upper to lower and vice versa
s.strip()Removes spaces from both ends
s.lstrip()Removes left spaces
s.rstrip()Removes right spaces
s.replace(old,new)Replaces substring
s.find(sub)Finds position of substring
s.index(sub)Finds position (error if not found)
s.count(sub)Counts occurrences
s.startswith(sub)Checks starting characters
s.endswith(sub)Checks ending characters
s.split()Splits string into list
'sep'.join(list)Joins list into string
s.isalpha()Checks all alphabets
s.isdigit()Checks all digits
s.isalnum()Checks alphanumeric
s.isspace()Checks only spaces
s.center(n)Centers string
s.zfill(n)Fills leading zeros
