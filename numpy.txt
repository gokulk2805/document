AUTHOR,PAGES,RATING
import numpy as np

n = int(input())

# Define structured array data type
dtype = [('Title', 'U50'),
         ('Author', 'U50'),
         ('Pages', 'i4'),
         ('Rating', 'f4')]

books_list = []

for _ in range(n):
    title, author, pages, rating = input().split()
    books_list.append((title, author, int(pages), float(rating)))

# Create structured array
books = np.array(books_list, dtype=dtype)

# Display all books
print("All books:")
print(books)

# Books with more than 300 pages
print("Books with more than 300 pages:")
print(books[books['Pages'] > 300])

# Books with rating greater than 4.0
print("Books with rating > 4.0:")
print(books[books['Rating'] > 4.0])

RIYA FILTERING SUBJECT
# You are using Python
import numpy as np

# Input rows and columns
R, C = map(int, input().split())

# Input marks
marks = []

for _ in range(R):
    row = list(map(int, input().split()))
    marks.append(row)

# Create NumPy array
arr = np.array(marks)

# Display all students' marks
print("Marks of all students:")
print(arr)

# Calculate average of each student
avg = np.mean(arr, axis=1)

# Format averages to 2 decimal places
formatted_avg = [f"{x:.2f}" for x in avg]

print("Average score of each student:")
print(formatted_avg)

# Filter students with average > 80
filtered = arr[avg > 80]

print("Students with average score above 80:")
print(filtered)
